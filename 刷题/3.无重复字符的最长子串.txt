def lengthOfLongestSubstring(self, s):
        """
        :type s: str
        :rtype: int
        """
        if len(s)<2: return len(s)
        left=0
        slide=set()
        slen=0
        imax=1
        for i in range(0,len(s)):
            slen+=1
            while s[i] in slide:
                slide.remove(s[left])
                left+=1
                slen-=1
            imax = slen if slen>imax else imax
            slide.add(s[i])
        return imax
